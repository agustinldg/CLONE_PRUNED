ESte es el codigo que genera el json que reicibira el grafico:
    WITH denorm AS (
      SELECT MEASURE_UTC,
          MAX(CASE WHEN ALIAS_ID = 'INDOOR' THEN TEMPERATURE END) INDOOR_TEMPERATURE,
          MAX(CASE WHEN ALIAS_ID = 'INDOOR' THEN HUMIDITY END) INDOOR_HUMIDITY,
          MAX(CASE WHEN ALIAS_ID = 'INDOOR' THEN CO2 END) INDOOR_CO2,
          MAX(CASE WHEN ALIAS_ID = 'INDOOR' THEN NOISE END) INDOOR_NOISE,
          MAX(CASE WHEN ALIAS_ID = 'INDOOR' THEN PRESSURE END) INDOOR_PRESSURE,
          MAX(CASE WHEN ALIAS_ID = 'OUTDOOR_1' THEN PRESSURE END) OUTDOOR_PRESSURE,
          MAX(CASE WHEN ALIAS_ID = 'OUTDOOR_2' THEN TEMPERATURE END) OUTDOOR_TEMPERATURE,
          MAX(CASE WHEN ALIAS_ID = 'OUTDOOR_2' THEN HUMIDITY END) OUTDOOR_HUMIDITY,
          CASE WHEN MEASURE_UTC >= l_end_date - 1 THEN 'D' ELSE 'B' END AS period
      FROM PULSI.NETAMO_MEASURES
      WHERE MEASURE_UTC BETWEEN l_end_date - 2 AND l_end_date
      GROUP BY MEASURE_UTC
    )
    SELECT JSON_OBJECT(
        'D' VALUE (SELECT JSON_ARRAYAGG(JSON_ARRAY(MEASURE_UTC, INDOOR_TEMPERATURE, INDOOR_HUMIDITY, INDOOR_CO2, INDOOR_NOISE, INDOOR_PRESSURE, OUTDOOR_PRESSURE, OUTDOOR_TEMPERATURE, OUTDOOR_HUMIDITY) ORDER BY MEASURE_UTC ASC RETURNING CLOB) FROM denorm WHERE period = 'D'),
        'B' VALUE (SELECT JSON_ARRAYAGG(JSON_ARRAY(MEASURE_UTC, INDOOR_TEMPERATURE, INDOOR_HUMIDITY, INDOOR_CO2, INDOOR_NOISE, INDOOR_PRESSURE, OUTDOOR_PRESSURE, OUTDOOR_TEMPERATURE, OUTDOOR_HUMIDITY) ORDER BY MEASURE_UTC ASC RETURNING CLOB) FROM denorm WHERE period = 'B')
        RETURNING CLOB
    ) INTO l_json_clob
    FROM dual;
